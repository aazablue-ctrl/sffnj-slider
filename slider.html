<script>

  const slides = document.querySelectorAll('.sffnj-slide');

  const prev = document.querySelector('.sffnj-prev');

  const next = document.querySelector('.sffnj-next');

  let index = 0;


  function showSlide(i) {

    slides.forEach((slide, idx) => {

      slide.classList.toggle('active', idx === i);

    });

  }


  prev.addEventListener('click', () => {

    index = (index - 1 + slides.length) % slides.length;

    showSlide(index);

  });


  next.addEventListener('click', () => {

    index = (index + 1) % slides.length;

    showSlide(index);

  });


  showSlide(index);

</script>
